function cylinder (r , L,h  )
for j=1:5
V(j)=L(j)*(r(j)^2*acos((r(j)-h(j))/r(j))-(r(j)-h(j))*sqrt(2*r(j)*h(j)-h(j)^2));

end
plot(h,V,'r-');
end

function Fn=Fnorm(A)
[n,m]=size(A);
sum=0;
for i=1:n
    for j=1:m
      sum=sum+A(i,j)^2;


    end
end
Fn=sqrt(sum);
end

        disp('D')

        disp('C')

elseif(jegy>=80 && jegy<90)
        disp('B')

elseif(jegy>=90 && jegy<=100)
        disp('A')

end
end
